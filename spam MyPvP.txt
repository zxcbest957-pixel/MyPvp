-- === 3. SMART REMOTE RESOLVER ===
local function getNetManaged()
    return ReplicatedStorage:WaitForChild("rbxts_include", 5):WaitForChild("node_modules", 5):WaitForChild("@rbxts", 5):WaitForChild("net", 5):WaitForChild("out", 5):WaitForChild("_NetManaged", 5)
end

local function getRemote(keyword)
    local net = getNetManaged()
    if not net then return nil end
    
    -- 1. Сначала ищем по точному совпадению
    local exact = net:FindFirstChild(keyword)
    if exact then return exact end

    -- 2. Если не нашли, ищем по ключевому слову (Fuzzy search)
    local lowerKey = keyword:lower():gsub(".*/", "") -- берем последнюю часть пути
    for _, v in pairs(net:GetChildren()) do
        if v.Name:lower():find(lowerKey) then
            print("[MyPvP] Found remote by fuzzy search: " .. v.Name .. " for key: " .. keyword)
            return v
        end
    end
    
    -- 3. Если совсем ничего, выводим список (для дебага)
    warn("[MyPvP] Could not find remote: " .. keyword)
    return nil
end

-- === 4. PRO-HUNTER ===
local function huntProPlayers()
    print("[MyPvP] INITIALIZING PRO-HUNTER...")
    local profileRemote = getRemote("Bedwars/GetPlayerProfile") or getRemote("GetPlayerProfile") or getRemote("Profile")
    local inviteRemote = getRemote("Clans/InvitePlayerToClan") or getRemote("InvitePlayerToClan") or getRemote("ClanInvite")
    
    if not profileRemote or not inviteRemote then
        print("[MyPvP] DEBUG: List of available remotes in _NetManaged:")
        local net = getNetManaged()
        if net then
            for _, v in pairs(net:GetChildren()) do print(" - " .. v.Name) end
        end
        return
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            task.spawn(function()
                local success, profile = pcall(function() return profileRemote:InvokeServer(player.UserId) end)
                if success and profile then
                    local wins = profile.wins or profile.totalWins or (profile.Data and profile.Data.wins) or 0
                    if wins >= MIN_WINS then
                        shared.MyPvP_Stats.ProsSeen = shared.MyPvP_Stats.ProsSeen + 1
                        print("[MyPvP] PRO: " .. player.Name .. " (" .. wins .. " wins)")
                        pcall(function()
                            inviteRemote:InvokeServer({userId = player.UserId, name = player.Name, displayName = player.DisplayName}, CLAN_ID)
                            shared.MyPvP_Stats.Invites = shared.MyPvP_Stats.Invites + 1
                        end)
                    end
                end
            end)
            task.wait(0.2)
        end
    end
end

-- === 5. CHAT & JUMP ===
local function send(text)
    pcall(function()
        local remote = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
        if remote and remote:FindFirstChild("SayMessageRequest") then
            remote.SayMessageRequest:FireServer(text, "All")
        else
            game:GetService("TextChatService").TextChannels.RBXGeneral:SendAsync(text)
        end
    end)
end

local function jump()
    print("[MyPvP] SERVER HOPPING...")
    if not table.find(shared.MyPvP_Session, game.JobId) then table.insert(shared.MyPvP_Session, game.JobId) end
    local function fetch()
        local url = "https://games.roblox.com/v1/games/"..LOBBY_ID.."/servers/Public?limit=100&excludeFullGames=true"
        local success, res = pcall(function() return game:HttpGet(url) end)
        if not success then return end
        local data = HttpService:JSONDecode(res)
        for _, s in pairs(data.data) do
            if s.id ~= game.JobId and not table.find(shared.MyPvP_Session, s.id) then return s.id end
        end
    end
    local target = fetch()
    if target then TeleportService:TeleportToPlaceInstance(LOBBY_ID, target, LocalPlayer) end
end

end)
