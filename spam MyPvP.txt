-- 3. УМНЫЙ ПОИСК РЕМОРТОВ
local function getRemote(name)
    local netManaged = ReplicatedStorage:WaitForChild("rbxts_include"):WaitForChild("node_modules"):WaitForChild("@rbxts"):WaitForChild("net"):WaitForChild("out"):WaitForChild("_NetManaged")
    
    -- Сначала ищем точное совпадение
    local exact = netManaged:FindFirstChild(name)
    if exact then return exact end
    
    -- Если не нашли, ищем по части имени (например, если Clans/GetMyClan называется GetMyClan)
    local partName = name:split("/")
    partName = partName[#partName] or name
    
    for _, v in pairs(netManaged:GetChildren()) do
        if v.Name:find(partName) then
            print("[Debug] Found remote by partial match: " .. v.Name)
            return v
        end
    end
    return nil
end

local function getNextQuote()
    shared.MyPvP_LastIndex = (shared.MyPvP_LastIndex % #Quotes) + 1
    local quote = Quotes[shared.MyPvP_LastIndex]
    local suffixes = {"", " !", " !!", " .", " ..", " ^"}
    return quote .. suffixes[math.random(1, #suffixes)]
end

-- 4. PRO-HUNTER
local function huntProPlayers()
    print("[Antigravity] Starting Pro-Hunter scan...")
    
    -- Пытаемся найти Clan ID из разных источников
    local myClanId = LocalPlayer:GetAttribute("ClanId") 
    
    if not myClanId then
        local getClanRemote = getRemote("Clans/GetMyClan") or getRemote("Clan/GetMyClan") or getRemote("GetMyClan")
        if getClanRemote then
            local success, data = pcall(function() return getClanRemote:InvokeServer() end)
            if success and data and data.clan then myClanId = data.clan.clanId end
        end
    end
    
    if not myClanId then
        warn("[Antigravity] CRITICAL: Could not find your Clan ID! Make sure you are in a clan.")
        return
    end
    
    print("[Antigravity] Clan ID: " .. tostring(myClanId))
    
    local profileRemote = getRemote("Bedwars/GetPlayerProfile") or getRemote("GetPlayerProfile")
    local inviteRemote = getRemote("Clans/InvitePlayerToClan") or getRemote("InvitePlayerToClan")
    
    if not (profileRemote and inviteRemote) then
        warn("[Antigravity] Could not find Invite or Profile remotes!")
        return
    end
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            task.spawn(function()
                local success, profile = pcall(function() return profileRemote:InvokeServer(player.UserId) end)
                if success and profile then
                    local wins = profile.wins or profile.totalWins or 0
                    if wins >= MIN_WINS then
                        print("[Pro-Hunter] Pro Found: " .. player.Name .. " (" .. wins .. " wins)")
                        local invSuccess, invRes = pcall(function()
                            return inviteRemote:InvokeServer({
                                userId = player.UserId,
                                name = player.Name,
                                displayName = player.DisplayName
                            }, myClanId)
                        end)
                        if invSuccess then print("[Pro-Hunter] ✅ Invite sent to " .. player.Name) end
                    end
                end
            end)
            task.wait(0.2)
        end
    end
    print("[Antigravity] Scan finished.")
end
