-- === 3. REMOTE UTILITIES ===
local function getNetManaged()
    local success, res = pcall(function()
        return ReplicatedStorage:WaitForChild("rbxts_include", 5):WaitForChild("node_modules", 5):WaitForChild("@rbxts", 5):WaitForChild("net", 5):WaitForChild("out", 5):WaitForChild("_NetManaged", 5)
    end)
    return success and res or nil
end

local function getRemote(name)
    local net = getNetManaged()
    if not net then return nil end
    
    -- Try direct path (handling names with slashes)
    local parts = name:split("/")
    local current = net
    for _, part in ipairs(parts) do
        current = current:FindFirstChild(part)
        if not current then break end
    end
    if current then return current end

    -- Try literal name match in children
    for _, v in pairs(net:GetChildren()) do
        if v.Name == name or v.Name:find(parts[#parts]) then
            return v
        end
    end
    return nil
end

local function dump(t)
    if type(t) ~= "table" then return tostring(t) end
    local s = "{ "
    for k, v in pairs(t) do s = s .. tostring(k) .. "=" .. tostring(v) .. ", " end
    return s .. "}"
end

local function huntProPlayers()
    print("[MyPvP] INITIALIZING PRO-HUNTER...")
    local profileRemote = getRemote("Bedwars/GetPlayerProfile") or getRemote("GetPlayerProfile")
    local inviteRemote = getRemote("Clans/InvitePlayerToClan") or getRemote("InvitePlayerToClan")
    
    if not profileRemote then warn("[MyPvP] CRITICAL: Profile Remote not found!") end
    if not inviteRemote then warn("[MyPvP] CRITICAL: Invite Remote not found!") end
    
    if not (profileRemote and inviteRemote) then return end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            task.spawn(function()
                local success, profile = pcall(function() return profileRemote:InvokeServer(player.UserId) end)
                if success and profile then
                    -- Debug: print("[MyPvP] Profile for " .. player.Name .. ": " .. dump(profile))
                    local wins = profile.wins or profile.totalWins or (profile.Data and profile.Data.wins) or 0
                    if wins >= MIN_WINS then
                        shared.MyPvP_Stats.ProsSeen = shared.MyPvP_Stats.ProsSeen + 1
                        print("[MyPvP] PRO DETECTED: " .. player.Name .. " (" .. wins .. " wins)")
                        
                        local invSuccess, invRes = pcall(function()
                            return inviteRemote:InvokeServer({
                                userId = player.UserId,
                                name = player.Name,
                                displayName = player.DisplayName
                            }, CLAN_ID)
                        end)
                        
                        if invSuccess then
                            shared.MyPvP_Stats.Invites = shared.MyPvP_Stats.Invites + 1
                            print("[MyPvP] ✅ SUCCESS: Invited " .. player.Name)
                        else
                            warn("[MyPvP] ❌ FAILED: Could not invite " .. player.Name .. " (Reason: " .. tostring(invRes) .. ")")
                        end
                    end
                end
            end)
            task.wait(0.2)
        end
    end
end

local function jump()
    print("[MyPvP] SERVER HOPPING...")
    if not table.find(shared.MyPvP_Session, game.JobId) then
        table.insert(shared.MyPvP_Session, game.JobId)
    end

    local function fetchServers(cursor)
        local url = "https://games.roblox.com/v1/games/"..LOBBY_ID.."/servers/Public?limit=100&excludeFullGames=true"
        if cursor then url = url .. "&cursor=" .. cursor end
        local success, res = pcall(function() return game:HttpGet(url) end)
        if not success then return nil end
        local data = HttpService:JSONDecode(res)
        if not data or not data.data then return nil end
        
        local possible = {}
        for _, s in pairs(data.data) do
            if s.id ~= game.JobId and not table.find(shared.MyPvP_Session, s.id) then
                table.insert(possible, s.id)
            end
        end
        
        if #possible > 0 then return possible[math.random(1, #possible)] end
        if data.nextPageCursor then return fetchServers(data.nextPageCursor) end
        return nil
    end

    local nextSrv = fetchServers()
    if nextSrv then
        TeleportService:TeleportToPlaceInstance(LOBBY_ID, nextSrv, LocalPlayer)
    else
        print("[MyPvP] No fresh servers found. Clearing cache and retrying...")
        shared.MyPvP_Session = {game.JobId}
        task.wait(3)
        jump()
    end
end
