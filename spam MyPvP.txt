-- 3. ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ
local function getRemote(name)
    return ReplicatedStorage:WaitForChild("rbxts_include"):WaitForChild("node_modules"):WaitForChild("@rbxts"):WaitForChild("net"):WaitForChild("out"):WaitForChild("_NetManaged"):WaitForChild(name)
end

local function send(text)
    pcall(function()
        local remote = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
        if remote and remote:FindFirstChild("SayMessageRequest") then
            remote.SayMessageRequest:FireServer(text, "All")
        else
            game:GetService("TextChatService").TextChannels.RBXGeneral:SendAsync(text)
        end
    end)
end

local function getNextQuote()
    shared.MyPvP_LastIndex = (shared.MyPvP_LastIndex % #Quotes) + 1
    local quote = Quotes[shared.MyPvP_LastIndex]
    local suffixes = {"", " !", " !!", " .", " ..", " ^"}
    return quote .. suffixes[math.random(1, #suffixes)]
end

-- 4. СИСТЕМА ПОИСКА ПРО-ИГРОКОВ (PRO-HUNTER)
local function huntProPlayers()
    print("[Antigravity] Starting Pro-Hunter scan...")
    
    local getClanRemote = getRemote("Clans/GetMyClan")
    local profileRemote = getRemote("Bedwars/GetPlayerProfile")
    local inviteRemote = getRemote("Clans/InvitePlayerToClan")
    
    if not (getClanRemote and profileRemote and inviteRemote) then
        warn("[Antigravity] One or more remotes not found! Scan aborted.")
        return
    end
    
    print("[Antigravity] Remotes found. Fetching clan data...")
    
    local success, myClanData = pcall(function() return getClanRemote:InvokeServer() end)
    if not success or not myClanData or not myClanData.clan or not myClanData.clan.clanId then 
        warn("[Antigravity] Clan check failed. Are you in a clan? Error: " .. tostring(myClanData))
        return 
    end
    
    local myClanId = myClanData.clan.clanId
    print("[Antigravity] Clan ID found: " .. tostring(myClanId))
    
    local players = Players:GetPlayers()
    print("[Antigravity] Scanning " .. #players .. " players...")
    
    for _, player in ipairs(players) do
        if player ~= LocalPlayer then
            task.spawn(function()
                local profileSuccess, profile = pcall(function() return profileRemote:InvokeServer(player.UserId) end)
                if profileSuccess and profile then
                    local wins = profile.wins or profile.totalWins or 0
                    if wins >= MIN_WINS then
                        print("[Pro-Hunter] Found Pro: " .. player.Name .. " (" .. wins .. " wins). Inviting...")
                        
                        local inviteSuccess, inviteResult = pcall(function()
                            return inviteRemote:InvokeServer({
                                userId = player.UserId,
                                name = player.Name,
                                displayName = player.DisplayName
                            }, myClanId)
                        end)

                        if inviteSuccess and (inviteResult == true or (type(inviteResult) == "table" and (inviteResult.success or not inviteResult.error))) then
                            print("[Pro-Hunter] ✅ INVITE SENT to " .. player.Name)
                        else
                            print("[Pro-Hunter] ⚠️ Failed/Declined for " .. player.Name)
                        end
                    end
                elseif not profileSuccess then
                    -- warn("[Debug] Failed to fetch profile for " .. player.Name)
                end
            end)
            task.wait(0.2)
        end
    end
    print("[Antigravity] Scan complete.")
end

-- 5. СИСТЕМА ПРЫЖКА
local function startJump()
    print("[Antigravity] Searching for a new lobby...")
    table.insert(shared.MyPvP_Session, game.JobId)

    local function findServer(cursor)
        local url = "https://games.roblox.com/v1/games/"..LOBBY_ID.."/servers/Public?limit=100&excludeFullGames=true"
        if cursor then url = url .. "&cursor=" .. cursor end
        local success, response = pcall(function() return game:HttpGet(url) end)
        if not success then return nil end
        local json = HttpService:JSONDecode(response)
        local pool = {}
        for _, s in pairs(json.data) do
            if s.id ~= game.JobId and not table.find(shared.MyPvP_Session, s.id) then table.insert(pool, s.id) end
        end
        if #pool > 0 then return pool[math.random(1, #pool)]
        elseif json.nextPageCursor then return findServer(json.nextPageCursor) end
        return nil
    end

    local target = findServer()
    if target then
        TeleportService:TeleportToPlaceInstance(LOBBY_ID, target, LocalPlayer)
    else
        shared.MyPvP_Session = {game.JobId}
        task.wait(2)
        startJump()
    end
end

        task.wait(WAIT_TIME)
    end

    -- Уходим на другой сервер
    task.wait(2)
    startJump()
end)
